41 child.SetAudioChannel(0, i);
42 child.name = _csound.availableAudioChannels[i];
43 var meter = go.GetComponent<Child3DMeter>();
44 _meters.Add(_csound.availableAudioChannels[i], meter);
45 var aS = go.GetComponent<AudioSource>();
46 // set doppler level to 0 to avoid artefacts when the camera moves
47 aS.dopplerLevel = 0;
48 aS.rolloffMode = AudioRolloffMode.Custom;
49 // when the audio listener is 'radius' meters far from the audio source, there will be no sound,
50 // since the rolloff function will lower the volume accordingly to the custom curve, and at maxDistance the volume will be 0.
51 // Let's add 'rollofTolerance' meters more to have some sound when the listener is equidistant from the created sources
52 aS.maxDistance = _radius + _rollofTolerance;
53 }
