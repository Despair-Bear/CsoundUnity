150 private void DrawCaption()
151 {
152 var infoText = "";
153 if (m_channelControllers != null)
154 {
155 //look for caption in channelControllers
156 bool captionFound = false;
157 for (int i = 0; i < m_channelControllers.arraySize && !captionFound; i++)
158 {
159 var cc = m_channelControllers.GetArrayElementAtIndex(i);
160 var prop = cc.FindPropertyRelative("type");
161 if (prop.stringValue.Contains("form"))
162 {
163 var cap = cc.FindPropertyRelative("caption");
164 infoText = cap.stringValue;
165 captionFound = true;
166 }
167 }
168 if (!captionFound) infoText = "No file selected";
169 }
170
171 EditorGUILayout.HelpBox(infoText, MessageType.None);
172 EditorGUILayout.LabelField("Csound file", m_csoundFileName.stringValue);
173 }

241 public void DropAreaGUI()
242 {
243 DefaultAsset obj = (DefaultAsset)m_csoundAsset.objectReferenceValue;
244 //
245 EditorGUI.BeginChangeCheck();
246 obj = (DefaultAsset)EditorGUILayout.ObjectField("Csd Asset", obj, typeof(DefaultAsset), false);
247
248 if (EditorGUI.EndChangeCheck())
249 {
250 Undo.RecordObject(target, "Set Csd");
251 // Debug.Log("selected new asset!");
252 if (obj == null ||
253 !AssetDatabase.GetAssetPath(obj).EndsWith(".csd", true, System.Globalization.CultureInfo.CurrentCulture))
254 {
255 // Debug.Log("asset is not valid, set Csd NULL");
256 SetCsd(null);
257 }
258 else
259 {
260 //Debug.Log("change asset, it is valid! setting csd");
261 if (AssetDatabase.TryGetGUIDAndLocalFileIdentifier(obj, out string guid, out long localId))
262 {
263 // Debug.Log("guid valid: " + guid + " loc " + localId);
264 SetCsd(guid);
265 }
266 else
267 {
268 Debug.LogWarning("GUID NOT FOUND");
269 SetCsd(null);
270 }
271 }
272 EditorUtility.SetDirty(csoundUnity.gameObject);
273 }

276 public void DrawChannelControllers()
277 {
278 m_drawChannels.boolValue = EditorGUILayout.Foldout(m_drawChannels.boolValue, "Control Channels", true);
279 if (m_drawChannels.boolValue)
280 {
281 if (m_channelControllers.arraySize < 1)
282 {
283 EditorGUILayout.HelpBox("No Control Channels available", MessageType.None);
284 return;
285 }
286
287 EditorGUILayout.HelpBox("Control Channels", MessageType.None);
288 if (m_channelControllers != null)
289 //create controls for each Csound channel found in the file descriptor
290 for (int i = 0; i < m_channelControllers.arraySize; i++)
291 {
292 var cc = m_channelControllers.GetArrayElementAtIndex(i);
293 var chanValue = cc.FindPropertyRelative("value");
294 var text = cc.FindPropertyRelative("text").stringValue;
295 var channel = cc.FindPropertyRelative("channel").stringValue;
296 string label = text.Length > 3 ? text : channel;
297 var type = cc.FindPropertyRelative("type").stringValue;
298
299 if (type.Contains("slider"))
300 {
301 var min = cc.FindPropertyRelative("min").floatValue;
302 var max = cc.FindPropertyRelative("max").floatValue;
303
304 EditorGUI.BeginChangeCheck();
305 chanValue.floatValue = EditorGUILayout.Slider(label, chanValue.floatValue, min, max);
306 if (EditorGUI.EndChangeCheck() && Application.isPlaying && csoundUnity != null)
307 {
308 csoundUnity.SetChannel(channel, chanValue.floatValue);
309 }
310 }
311 else if (type.Contains("combobox"))
312 {
313 EditorGUI.BeginChangeCheck();
314 var options = cc.FindPropertyRelative("options");
315 var strings = new string[options.arraySize];
316 for (var s = 0; s < strings.Length; s++) {
317 strings[s] = options.GetArrayElementAtIndex(s).stringValue;
318 }
319 chanValue.floatValue = EditorGUILayout.Popup((int)chanValue.floatValue, strings);
320 if (EditorGUI.EndChangeCheck() && Application.isPlaying && csoundUnity != null)
321 {
322 csoundUnity.SetChannel(channel, chanValue.floatValue + 1);
323 }
324 }
325 else if (type.Contains("button"))
326 {
327 if (GUILayout.Button(label))
328 {
329 if (Application.isPlaying && csoundUnity != null)
330 {
331 chanValue.floatValue = chanValue.floatValue == 1 ? 0 : 1;
332 csoundUnity.SetChannel(channel, chanValue.floatValue);
333 }
334 }
335 }
336 else if (type.Contains("groupbox"))
337 {
338 EditorGUILayout.HelpBox(text, MessageType.None);
339 }
340 else if (type.Contains("checkbox"))
341 {
342 EditorGUI.BeginChangeCheck();
343 chanValue.floatValue = EditorGUILayout.Toggle(label, chanValue.floatValue == 1 ? true : false) ? 1f : 0f;
344 if (EditorGUI.EndChangeCheck() && Application.isPlaying && csoundUnity != null)
345 {
346 csoundUnity.SetChannel(channel, chanValue.floatValue);
347 }
348 }
349 }
