162 void ProcessBlock(float[] samples, int numChannels)
163 {
164 // print("CsoundUnityChild DSP Time - " + AudioSettings.dspTime * 48000);
165 if (availableAudioChannels == null || availableAudioChannels.Count < 1 || !csoundUnity.IsInitialized)
166 {
167 return;
168 }
169
170 for (int i = 0; i < (int)AudioChannelsSetting; i++)
171 {
172 var chanToUse = availableAudioChannels[selectedAudioChannelIndexByChannel[i]];
173 if (string.IsNullOrWhiteSpace(chanToUse)) continue;
174 if (!csoundUnity.namedAudioChannelDataDict.ContainsKey(chanToUse)) continue;
175 namedAudioChannelData[i] = csoundUnity.namedAudioChannelDataDict[chanToUse];
176 }
177
178 for (int i = 0, sampleIndex = 0; i < samples.Length; i += numChannels, sampleIndex++)
179 {
180 for (uint channel = 0; channel < numChannels; channel++)
181 {
182 switch (AudioChannelsSetting)
183 {
184 case AudioChannels.MONO:
185 // sample is multiplied by 0.5f to obtain the same volume as the original audio file,
186 // since the mono channel is duplicated between the channels
187 samples[i + channel] = samples[i + channel] * (float)(namedAudioChannelData[0][sampleIndex] / zerodbfs * 0.5f);
188 break;
189 case AudioChannels.STEREO:
190 samples[i + channel] = samples[i + channel] * (float)(namedAudioChannelData[(int)channel][sampleIndex] / zerodbfs);
191 break;
192 }
193 }
194 }
195 }
