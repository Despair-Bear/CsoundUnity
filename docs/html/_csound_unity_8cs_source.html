335 if (csound != null)
336 {
338 if (channels != null)
339 // initialise channels if found in xml descriptor..
340 for (int i = 0; i < channels.Count; i++)
341 {
342 if (channels[i].type.Contains("combobox"))
343 csound.SetChannel(channels[i].channel, channels[i].value + 1);
344 else
345 csound.SetChannel(channels[i].channel, channels[i].value);
346 // update channels index dictionary
347 if (!_channelsIndexDict.ContainsKey(channels[i].channel))
348 _channelsIndexDict.Add(channels[i].channel, i);
349 }
350
351 foreach (var name in availableAudioChannels)
352 {
353 if (!namedAudioChannelDataDict.ContainsKey(name))
354 {
355 namedAudioChannelDataDict.Add(name, new MYFLT[bufferSize]);
356 namedAudioChannelTempBufferDict.Add(name, new MYFLT[ksmps]);
357 }
358 }
359
361 LoggingCoroutine = StartCoroutine(Logging(.01f));
362
363 compiledOk = csound.CompiledWithoutError();
364
365 if (compiledOk)
366 {
367 zerdbfs = (float)csound.Get0dbfs();
368
369 Debug.Log($"Csound zerdbfs: {zerdbfs}");
370
371 initialized = true;
372 OnCsoundInitialized?.Invoke();
373 }
374 }
375 else
376 {
377 Debug.Log("Error creating Csound object");
378 compiledOk = false;
379 }
380
381 Debug.Log($"CsoundUnity done init, compiledOk? {compiledOk}");
382 }

565 public static List<string> ParseCsdFileForAudioChannels(string filename)
566 {
567 if (!File.Exists(filename)) return null;
568
569 string[] fullCsdText = File.ReadAllLines(filename);
570 if (fullCsdText.Length < 1) return null;
571
572 List<string> locaAudioChannels = new List<string>();
573
574 foreach (string line in fullCsdText)
575 {
576 var trimmd = line.TrimStart();
577 if (!trimmd.Contains("chnset")) continue;
578 if (trimmd.StartsWith(";")) continue;
579 var lndx = trimmd.IndexOf("chnset");
580 var chnsetEnd = lndx + "chnset".Length + 1;
581 var prms = trimmd.Substring(chnsetEnd, trimmd.Length - chnsetEnd);
582 var split = prms.Split(',');
583 if (!split[0].StartsWith("a") && !split[0].StartsWith("ga"))
584 continue; //discard non audio variables
585 // Debug.Log("found audio channel");
586 var ach = split[1].Replace('\\', ' ').Replace('\"', ' ').Trim();
587 if (!locaAudioChannels.Contains(ach))
588 locaAudioChannels.Add(ach);
589 }
590 return locaAudioChannels;
591 }

598 public static List<CsoundChannelController> ParseCsdFile(string filename)
599 {
600 if (!File.Exists(filename)) return null;
601
602 string[] fullCsdText = File.ReadAllLines(filename);
603 if (fullCsdText.Length < 1) return null;
604
605 List<CsoundChannelController> locaChannelControllers;
606 locaChannelControllers = new List<CsoundChannelController>();
607
608 foreach (string line in fullCsdText)
609 {
610
611 if (line.Contains("</"))
612 break;
613
614 var trimmd = line.TrimStart();
615 //discard csound comments in cabbage widgets
616 if (trimmd.StartsWith(";"))
617 {
618 //Debug.Log("discarding "+line);
619 continue;
620 }
621 string newLine = trimmd;
622 string control = trimmd.Substring(0, trimmd.IndexOf(" ") > -1 ? trimmd.IndexOf(" ") : 0);
623 if (control.Length > 0)
624 newLine = newLine.Replace(control, "");
625
626 if (control.Contains("slider") || control.Contains("button") || control.Contains("checkbox")
627 || control.Contains("groupbox") || control.Contains("form") || control.Contains("combobox"))
628 {
629 CsoundChannelController controller = new CsoundChannelController();
630 controller.type = control;
631
632 if (trimmd.IndexOf("caption(") > -1)
633 {
634 string infoText = trimmd.Substring(trimmd.IndexOf("caption(") + 9);
635 infoText = infoText.Substring(0, infoText.IndexOf(")") - 1);
636 controller.caption = infoText;
637 }
638
639 if (trimmd.IndexOf("text(") > -1)
640 {
641 string text = trimmd.Substring(trimmd.IndexOf("text(") + 6);
642 text = text.Substring(0, text.IndexOf(")") - 1);
643 text = text.Replace("\"", "");
644 text = text.Replace('"', new char());
645 if (controller.type == "combobox") //if combobox, create a range
646 {
647 char[] delimiterChars = { ',' };
648 string[] tokens = text.Split(delimiterChars);
649 controller.SetRange(1, tokens.Length, 0);
650
651 for (var o = 0; o < tokens.Length; o++)
652 {
653 tokens[o] = string.Join("", tokens[o].Split(default(string[]), System.StringSplitOptions.RemoveEmptyEntries));
654 }
655 controller.options = tokens;
656 }
657 }
658
659 if (trimmd.IndexOf("items(") > -1)
660 {
661 string text = trimmd.Substring(trimmd.IndexOf("items(") + 7);
662 text = text.Substring(0, text.IndexOf(")") - 1);
663 //TODO THIS OVERRIDES TEXT!
664 text = text.Replace("\"", "");
665 text = text.Replace('"', new char());
666 if (controller.type == "combobox")
667 {
668 char[] delimiterChars = { ',' };
669 string[] tokens = text.Split(delimiterChars);
670 controller.SetRange(1, tokens.Length, 0);
671
672 for (var o = 0; o < tokens.Length; o++)
673 {
674 tokens[o] = string.Join("", tokens[o].Split(default(string[]), System.StringSplitOptions.RemoveEmptyEntries));
675 }
676 controller.options = tokens;
677 }
678 }
679
680 if (trimmd.IndexOf("channel(") > -1)
681 {
682 string channel = trimmd.Substring(trimmd.IndexOf("channel(") + 9);
683 channel = channel.Substring(0, channel.IndexOf(")") - 1);
684 controller.channel = channel;
685 }
686
687 if (trimmd.IndexOf("range(") > -1)
688 {
689 int rangeAt = trimmd.IndexOf("range(");
690 if (rangeAt != -1)
691 {
692 string range = trimmd.Substring(rangeAt + 6);
693 range = range.Substring(0, range.IndexOf(")"));
694 char[] delimiterChars = { ',' };
695 string[] tokens = range.Split(delimiterChars);
696 for (var i = 0; i < tokens.Length; i++)
697 {
698 tokens[i] = string.Join("", tokens[i].Split(default(string[]), StringSplitOptions.RemoveEmptyEntries));
699 if (tokens[i].StartsWith("."))
700 {
701 tokens[i] = "0" + tokens[i];
702 }
703 if (tokens[i].StartsWith("-."))
704 {
705 tokens[i] = "-0" + tokens[i].Substring(2, tokens[i].Length - 2);
706 }
707 }
708 var min = float.Parse(tokens[0], CultureInfo.InvariantCulture);
709 var max = float.Parse(tokens[1], CultureInfo.InvariantCulture);
710 var val = float.Parse(tokens[2], CultureInfo.InvariantCulture);
711 controller.SetRange(min, max, val);
712 }
713 }
714
715 if (line.IndexOf("value(") > -1)
716 {
717 string value = trimmd.Substring(trimmd.IndexOf("value(") + 6);
718 value = value.Substring(0, value.IndexOf(")"));
719 value = value.Replace("\"", "");
720 controller.value = value.Length > 0 ? float.Parse(value, CultureInfo.InvariantCulture) : 0;
721 if (control.Contains("combobox"))
722 {
723 //Cabbage combobox index starts from 1
724 controller.value = controller.value - 1;
725 // Debug.Log("combobox value in parse: " + controller.value);
726 }
727 }
728 locaChannelControllers.Add(controller);
729 }
730 }
731 return locaChannelControllers;
732 }

1135 public static float Remap(float value, float from1, float to1, float from2, float to2, bool clamp = false)
1136 {
1137 float retValue = (value - from1) / (to1 - from1) * (to2 - from2) + from2;
1138 if (float.IsNaN(retValue)) return from2;
1139 if (float.IsPositiveInfinity(retValue)) return to2;
1140 if (float.IsNegativeInfinity(retValue)) return from2;
1141 return clamp ? Mathf.Clamp(retValue, from2, to2) : retValue;
1142 }

1168 public static MYFLT[] GetSamples(string source, SamplesOrigin origin, int channelNumber = 1, bool writeChannelData = false)
1169 {
1170 MYFLT[] res = new MYFLT[0];
1171 switch (origin)
1172 {
1173 case SamplesOrigin.Resources:
1174 var src = Resources.Load<AudioClip>(source);
1175 if (src == null)
1176 {
1177 res = null;
1178 break;
1179 }
1180 var data = new float[src.samples * src.channels];
1181 src.GetData(data, 0);
1182
1183 if (writeChannelData)
1184 {
1185 res = new MYFLT[src.samples * src.channels + 1];

1423 private void ProcessBlock(float[] samples, int numChannels)
1424 {
1425 if (compiledOk && initialized && !_quitting)
1426 {
1427 for (int i = 0; i < samples.Length; i += numChannels, ksmpsIndex++)
1428 {
1429 for (uint channel = 0; channel < numChannels; channel++)
1430 {
1431 // necessary to avoid calling csound functions when quitting while reading this block of samples
1432 // always remember OnAudioFilterRead runs on a different thread
1433 if (_quitting) return;
1434
1435 if (mute == true)
1436 samples[i + channel] = 0.0f;
1437 else
1438 {
1439 if ((ksmpsIndex >= GetKsmps()) && (GetKsmps() > 0))
1440 {
1441 var res = PerformKsmps();
1442 performanceFinished = res == 1;
1443 ksmpsIndex = 0;
1444
1445 foreach (var chanName in availableAudioChannels)
1446 {
1447 if (!namedAudioChannelTempBufferDict.ContainsKey(chanName)) continue;
1448 namedAudioChannelTempBufferDict[chanName] = GetAudioChannel(chanName);
1449 }
1450 }
1451
1452 if (processClipAudio)
1453 {
1454 SetInputSample((int)ksmpsIndex, (int)channel, samples[i + channel] * zerdbfs);
1455 }
1456
1457 //if csound nChnls are more than the current channel, set the last csound channel available on the sample (assumes GetNchnls above 0)
1458 var outputSampleChannel = channel < GetNchnls() ? channel : GetNchnls() - 1;
1459 var output = (float)GetOutputSample((int)ksmpsIndex, (int)outputSampleChannel) / zerdbfs;
1460 // multiply Csound output by the sample value to maintain spatialization set by Unity.
1461 // don't multiply if reading from a clip: this should maintain the spatialization of the clip anyway
1462 samples[i + channel] = processClipAudio ? output : samples[i + channel] * output;
1463
1464 if (loudVolumeWarning && (samples[i + channel] > loudWarningThreshold))
1465 {
1466 samples[i + channel] = 0.0f;
1467 Debug.LogWarning("Volume is too high! Clearing output");
1468 }
1469 }
1470 }
1471
1472 // update the audioChannels just when this instance is not muted
1473 if (!mute)
1474 foreach (var chanName in availableAudioChannels)
1475 {
1476 if (!namedAudioChannelDataDict.ContainsKey(chanName) || !namedAudioChannelTempBufferDict.ContainsKey(chanName)) continue;
1477 namedAudioChannelDataDict[chanName][i / numChannels] = namedAudioChannelTempBufferDict[chanName][ksmpsIndex];
1478 }
1479 }
1480 }
