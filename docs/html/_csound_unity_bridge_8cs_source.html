47 private void SetEnvironmentSettings(List<EnvironmentSettings> environmentSettings)
48 {
49 if (environmentSettings == null || environmentSettings.Count == 0) return;
50 foreach (var env in environmentSettings)
51 {
52 switch (Application.platform)
53 {
54 case RuntimePlatform.OSXEditor:
55 case RuntimePlatform.OSXPlayer:
56 if (env.platform.Equals(SupportedPlatform.MacOS))
57 {
58 Debug.Log($"Setting {env.GetTypeString()} for MacOS to: {env.GetPath()}");
59 Csound6.NativeMethods.csoundSetGlobalEnv(env.GetTypeString(), env.GetPath());
60 }
61 break;
62 case RuntimePlatform.WindowsPlayer:
63 case RuntimePlatform.WindowsEditor:
64 if (env.platform.Equals(SupportedPlatform.Windows))
65 {
66 Debug.Log($"Setting {env.GetTypeString()} for Windows to: {env.GetPath()}");
67 Csound6.NativeMethods.csoundSetGlobalEnv(env.GetTypeString(), env.GetPath());
68 }
69 break;
70 case RuntimePlatform.Android:
71 if (env.platform.Equals(SupportedPlatform.Android))
72 {
73 Debug.Log($"Setting {env.GetTypeString()} for Android to: {env.GetPath()}");
74 Csound6.NativeMethods.csoundSetGlobalEnv(env.GetTypeString(), env.GetPath());
75 }
76 break;
77 case RuntimePlatform.IPhonePlayer:
78 if (env.platform.Equals(SupportedPlatform.iOS))
79 {
80 Debug.Log($"Setting {env.GetTypeString()} for iOS to: {env.GetPath()}");
81 Csound6.NativeMethods.csoundSetGlobalEnv(env.GetTypeString(), env.GetPath());
82 }
83 break;
84 default:
85 break;
86 }
87 }

101 // KEEP THIS FOR REFERENCE ;)
102 //#if (UNITY_EDITOR_WIN || UNITY_STANDALONE_WIN)

540 public IDictionary<string, IList<OpcodeArgumentTypes>> GetOpcodeList()
541 {
542 var opcodes = new SortedDictionary<string, IList<OpcodeArgumentTypes>>();
543 IntPtr ppOpcodeList = IntPtr.Zero;
544 int size = Csound6.NativeMethods.csoundNewOpcodeList(csound, out ppOpcodeList);
545 if ((ppOpcodeList != IntPtr.Zero) && (size >= 0))
546 {
547 int proxySize = Marshal.SizeOf(typeof(OpcodeListProxy));
548 for (int i = 0; i < size; i++)
549 {
550 OpcodeListProxy proxy = Marshal.PtrToStructure(ppOpcodeList + (i * proxySize), typeof(OpcodeListProxy)) as OpcodeListProxy;
551 string opname = Marshal.PtrToStringAnsi(proxy.opname);
552 OpcodeArgumentTypes opcode = new OpcodeArgumentTypes
553 {
554 outypes = Marshal.PtrToStringAnsi(proxy.outtypes),
555 intypes = Marshal.PtrToStringAnsi(proxy.intypes),
556 flags = proxy.flags
557 };
558 if (!opcodes.ContainsKey(opname))
559 {
560 IList<OpcodeArgumentTypes> types = new List<OpcodeArgumentTypes>();
561 types.Add(opcode);
562 opcodes.Add(opname, types);
563 }
564 else
565 {
566 opcodes[opname].Add(opcode);
567 }
568 }
569 Csound6.NativeMethods.csoundDisposeOpcodeList(csound, ppOpcodeList);
570 }
571 return opcodes;
572 }

581 public IDictionary<string, ChannelInfo> GetChannelList()
582 {
583 IDictionary<string, ChannelInfo> channels = new SortedDictionary<string, ChannelInfo>();
584
585 IntPtr ppChannels = IntPtr.Zero;
586 int size = Csound6.NativeMethods.csoundListChannels(csound, out ppChannels);
587 if ((size > 0) && (ppChannels != IntPtr.Zero))
588 {
589 int proxySize = Marshal.SizeOf(typeof(ChannelInfoProxy));
590 for (int i = 0; i < size; i++)
591 {
592 var proxy = Marshal.PtrToStructure(ppChannels + (i * proxySize), typeof(ChannelInfoProxy)) as ChannelInfoProxy;
593 string chanName = CharPtr2String(proxy.name);
594
595 ChannelInfo info = new ChannelInfo(chanName, (ChannelType)(proxy.type & 15), (ChannelDirection)(proxy.type >> 4));
596 var hintProxy = proxy.hints;
597 var hints = new ChannelHints((ChannelBehavior)hintProxy.behav, hintProxy.dflt, hintProxy.min, hintProxy.max)
598 {
599 x = hintProxy.x,
600 y = hintProxy.y,
601 height = hintProxy.height,
602 width = hintProxy.width,
603 attributes = CharPtr2String(proxy.name)
604 };
605 info.Hints = hints;
606 channels.Add(chanName, info);
607 }
608 Csound6.NativeMethods.csoundDeleteChannelList(csound, ppChannels);
609 }
610 return channels;
611 }
